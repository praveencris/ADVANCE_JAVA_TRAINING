package demo;

import java.lang.annotation.ElementType;
import java.lang.annotation.Retention;
import java.lang.annotation.RetentionPolicy;
import java.lang.annotation.Target;
import java.lang.annotation.Annotation;

@Retention(RetentionPolicy.RUNTIME)
@Target(ElementType.TYPE)
@interface TestInfo{

	public enum Priority {
	   LOW, MEDIUM, HIGH
	}

	
	
	String tags() default "";
	
	String createdBy() default "sdf";
	
	String lastModified() default "03/01/2014";

}








@TestInfo(tags="html", createdBy="Dimple",
lastModified="04/04/2010")
public class TestDemo
{
	Class<TestDemo> obj = TestDemo.class;

	public void welcome()
	{

	

	Annotation annotation = obj.getAnnotation
			(TestInfo.class);
	
		TestInfo testerInfo = (TestInfo) 
				annotation;
	System.out.println("Annotation value"
				+testerInfo.createdBy()+"   "
			+testerInfo.lastModified());		
//String create=testerInfo.createdBy();
	
	}
	public static void main(String args[])
	{
		TestDemo t=new TestDemo();
		t.welcome();
	}		
}
